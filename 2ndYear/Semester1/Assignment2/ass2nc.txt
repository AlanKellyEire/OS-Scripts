#!/bin/bash# Assignment 2 Alan Kelly r00052131 DNet2if (( "${#}" != 2))then	echo "${0}: ERROR: Incorrect number of arguments entered." 1>&2	echo "${0}: USAGE: ${0} <source directory> <new destination directory>" 1>&2	exit 8fiif [[ ! -d "${1}" ]]then	echo "${0}: ERROR: directory ${1} does not exist." 1>&2	echo "${0}: USAGE: ${0} <source directory> <new destination directory>" 1>&2	exit 7fiif test  -d "${2}"then	echo "${0}: ERROR: directory ${2} already exists." 1>&2	echo "${0}: USAGE: ${0} <source directory> <new destination directory>" 1>&2	exit 6fiif test -f "${2}"then	echo "${0}:ERROR:${2} is a file not a directory" 1>&2	echo "${0}:USAGE:Enter the name of an existing directory for the destination ." 1>&2	exit 5fiif  test ! -e "${2}"then	if ! mkdir "${2}" 2>/dev/null 	then 		echo "${0}:ERROR:Failed to create ${2}" 1>&2		echo "${0}:INFO:Check permissions or disk space." 1>&2		exit 4	fifitypeset -i success=0typeset -i fail=0for f in $(ls "${1}")do	if [[ -f "${1}/${f}" ]]	then 		if cp ${1}/${f} ${2}		then			((success++))		else 			echo "${0}:WARNING:File ${f} could not be copied to ${2}" 1>&2			((fail++))		fi		fidoneif (( fail == 0 && success == 0 ))then	echo "${0}:ERROR: ${1} does not contain ordinary files" 1>&2	echo "${0}:USAGE: Please enter a directory thats contains ordinary files!!!" 1>&2		exit 3fiif (( fail == 0 && success != 0 ))then	echo "${0}:All ordinary files from ${1} have been copied to ${2}" 1>&2	exit 2fiif (( fail != 0 && success == 0 )) then	echo "${0}:No ordinary files from ${1} have been copied to ${2}!!!!" 1>&2	exit 1fiif (( fail != 0 && success != 0 ))then	echo "${0}:Some ordinary files from ${1} have been copied to ${2}!!!!" 1>&2 	exit 0fi
